# Funkcja normalizujaca metoda min - max
normalize <- function(x) {
  if (min(x) == max(x)) {
    return(rep(0, length(x)))
  } else {
    return((x - min(x)) / (max(x) - min(x)))
  }
}

  mutate(across(where(is.numeric), normalize))
